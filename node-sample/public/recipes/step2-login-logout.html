}

function login() {
    $.ajax({    
        type: "POST",
        url: myDHS.app_token_url,
        data: {app_scope: 'ACCOUNT_ID'},
        success: function(data) {
            accessToken = JSON.parse(data);
            associateAccessToken();
        },
        error: function (data) {
            console.log(data)
        }
    });
}
  
function associateAccessToken() {
    phone.associateAccessToken({
        userId: loginInput.value,
        token: accessToken.access_token,
        success: function () {
            phone.login({token: accessToken.access_token});
        },
        error: onError
    }); 
}
    
function logout() {
    phone.logout();
}
